import pyodbc

# 数据库连接字符串
conn_str = (
    r'Driver={Microsoft Access Driver (*.mdb, *.accdb)};'
    r'DBQ=.\jiaoxueguanli.accdb;'  # 你的数据库文件地址，这里为本目录下
)

try:
    # 连接数据库
    conn = pyodbc.connect(conn_str)
    cursor = conn.cursor()

    # SQL查询语句
    SQL_value = '''
        SELECT 
            student.姓名, 
            courseChoose.学期, 
            course.课程名称, 
            course.课程号, 
            courseChoose.分数, 
            teacher.姓名, 
            class.班级名称, 
            class.所属院系 
        FROM 
            student 
            INNER JOIN courseChoose ON student.学号 = courseChoose.学号 
            INNER JOIN course ON courseChoose.课程号 = course.课程号 
            INNER JOIN class ON student.班级编号 = class.班级编号 
            INNER JOIN teacher ON course.教师编号 = teacher.教师编号 
    '''

    # 执行查询
    cursor.execute(SQL_value)

    # 获取查询结果
    result = cursor.fetchall()

    # 输出查询结果
    for row in result:
        for i in range(8):
            print(row[i], "  ", end="")
        print()

except pyodbc.Error as e:
    print(f"数据库操作出错：{e}")

finally:
    # 关闭游标和连接
    if 'cursor' in locals():
        cursor.close()
    if 'conn' in locals():
        conn.close()
